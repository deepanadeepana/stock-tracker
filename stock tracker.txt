import requests

API_KEY = 'YOUR_ALPHA_VANTAGE_API_KEY'  # Replace with your key from alphavantage.co
portfolio = {}


def get_stock_price(symbol):
    url = f'https://www.alphavantage.co/query?function=GLOBAL_QUOTE&symbol={symbol}&apikey={API_KEY}'
    response = requests.get(url)
    data = response.json()
    try:
        return float(data['Global Quote']['05. price'])
    except:
        return None

def display_portfolio():
    print("\nüìä Your Portfolio:")
    total_value = 0
    total_profit_loss = 0
    for symbol, stock in portfolio.items():
        shares = stock['shares']
        buy_price = stock['buy_price']
        current_price = get_stock_price(symbol)

        if current_price:
            value = shares * current_price
            profit_loss = (current_price - buy_price) * shares
            total_value += value
            total_profit_loss += profit_loss

            print(f"\n{symbol}:")
            print(f"  Shares: {shares}")
            print(f"  Buy Price: ‚Çπ{buy_price:.2f}")
            print(f"  Current Price: ‚Çπ{current_price:.2f}")
            print(f"  Value: ‚Çπ{value:.2f}")
            print(f"  ‚û§ {'Profit' if profit_loss >= 0 else 'Loss'}: ‚Çπ{profit_loss:.2f}")
        else:
            print(f"\n{symbol}: Could not retrieve price.")

    print(f"\nüíº Total Portfolio Value: ‚Çπ{total_value:.2f}")
    print(f"üìà Total Profit/Loss: ‚Çπ{total_profit_loss:.2f}\n")

def main():
    while True:
        print("=== üìà Stock Portfolio Tracker ===")
        print("1. Add Stock")
        print("2. Remove Stock")
        print("3. View Portfolio")
        print("4. Exit")

        choice = input("Enter choice (1-4): ")

        if choice == '1':
            symbol = input("Enter Stock Symbol (e.g., AAPL): ").upper()
            try:
                shares = int(input("Enter number of shares: "))
                buy_price = float(input("Enter buy price per share: ‚Çπ"))
            except ValueError:
                print("‚ö†Ô∏è Invalid input. Try again.\n")
                continue

            if get_stock_price(symbol):
                portfolio[symbol] = {'shares': shares, 'buy_price': buy_price}
                print(f"‚úÖ {shares} shares of {symbol} added.\n")
            else:
                print("‚ùå Invalid stock symbol or API error.\n")

        elif choice == '2':
            symbol = input("Enter Stock Symbol to remove: ").upper()
            if symbol in portfolio:
                del portfolio[symbol]
                print(f"üóëÔ∏è {symbol} removed from portfolio.\n")
            else:
                print("‚ö†Ô∏è Stock not found.\n")

        elif choice == '3':
            display_portfolio()

        elif choice == '4':
            print("üëã Exiting. Goodbye!")
            break

        else:
            print("‚ùå Invalid choice. Try again.\n")

if __name__ == "__main__":
    main()
